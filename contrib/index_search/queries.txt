

Simple search:
	SELECT User.*
	BASE 'dc=gonicus,dc=de' SUB

	--

	SELECT User.dn, User.objectClass 
	BASE 'dc=gonicus,dc=de' SUB
	WHERE cn="*";

Search by parent container (Condition):
	SELECT User.dn, User.objectClass 
	BASE 'dc=gonicus,dc=de' SUB
	WHERE cn="*" AND <parent_container>="ou=people,";

Search combines (Condition):
	SELECT User.dn, User.objectClass, Group.cn 
	BASE 'dc=gonicus,dc=de' SUB
	WHERE User.gidNumber=Group.gidNumber AND Group.desc != "";

Search with ordered and limited results:
	SELECT User.dn, User.objectClass
	BASE 'dc=gonicus,dc=de' SUB
	ORDER BY User.sn, User.givenName
	LIMIT 10, 20
